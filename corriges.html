        essai = int(essai_str)

            print "C'est plus !"
        elif essai > nombre_secret:
            print "C'est moins !"
        else:
            print "Bravo ! Vous avez trouve le nombre secret ({0}) en {1} essais.".format(nombre_secret, nombre_essais)
            break # On sort de la boucle while car le jeu est fini
            
    except ValueError:

# -*- coding: utf-8 -*-

def celsius_vers_fahrenheit(temp_celsius):
    """
    Convertit une temperature de degres Celsius en degres Fahrenheit.
    Formule : F = C * 9/5 + 32
    """
    temp_fahrenheit = temp_celsius * 9.0/5.0 + 32
    return temp_fahrenheit

# --- Programme principal ---
try:
    celsius_str = raw_input("Entrez une temperature en degres Celsius : ")
    celsius = float(celsius_str)
    
    fahrenheit = celsius_vers_fahrenheit(celsius)
    
    print "{0:.1f}¬∞C equivaut a {1:.1f}¬∞F".format(celsius, fahrenheit)

except ValueError:
    print "Erreur : Veuillez entrer une temperature valide."

# -*- coding: utf-8 -*-

def afficher_liste(courses):
    """Affiche le contenu de la liste de courses."""
    print "\n--- MA LISTE DE COURSES ---"
    if not courses:
        print "La liste est vide."
    else:
        for i, item in enumerate(courses):
            print "{0}. {1}".format(i + 1, item)
    print "-------------------------"

# --- Programme principal ---
ma_liste = []

while True:
    afficher_liste(ma_liste)
    choix = raw_input("Que voulez-vous faire ? (ajouter, supprimer, quitter) : ").lower()

    if choix == "quitter":
        break
    elif choix == "ajouter":
        nouvel_item = raw_input("Quel item ajouter ? : ")
        ma_liste.append(nouvel_item)
        print "'{0}' a ete ajoute a la liste.".format(nouvel_item)
    elif choix == "supprimer":
        try:
            num_item_str = raw_input("Quel numero d'item supprimer ? : ")
            num_item = int(num_item_str)
            
            # On retire 1 car l'utilisateur voit une liste numerotee a partir de 1
            if 1 <= num_item <= len(ma_liste):
                item_supprime = ma_liste.pop(num_item - 1)
                print "'{0}' a ete supprime.".format(item_supprime)
            else:
                print "Ce numero n'est pas dans la liste."
        except ValueError:
            print "Veuillez entrer un numero valide."
    else:
        print "Choix non valide."

print "Au revoir !"

Corrig√© : Gestionnaire de t√¢ches en ligne de commande

Ce corrig√© impl√©mente une des options du projet final. Il combine les fonctions, les listes, la gestion des erreurs et la manipulation de fichiers.

# -*- coding: utf-8 -*-

NOM_FICHIER = "taches.txt"

def charger_taches():
    """Charge les taches depuis le fichier taches.txt."""
    try:
        with open(NOM_FICHIER, "r") as f:
            # .strip() enleve les espaces et sauts de ligne
            # On ne garde que les lignes qui ne sont pas vides
            taches = [ligne.strip() for ligne in f.readlines() if ligne.strip()]
    except IOError:
        # Si le fichier n'existe pas, on retourne une liste vide
        taches = []
    return taches

def sauvegarder_taches(taches):
    """Sauvegarde la liste des taches dans le fichier."""
    with open(NOM_FICHIER, "w") as f:
        for tache in taches:
            f.write(tache + "\n")

def afficher_taches(taches):
    """Affiche la liste des taches."""
    print "\n--- LISTE DES TACHES ---"
    if not taches:
        print "Aucune tache pour le moment."
    else:
        for i, tache in enumerate(taches):
            print "{0}. {1}".format(i + 1, tache)
    print "------------------------"

# --- Programme Principal ---
taches = charger_taches()

while True:
    afficher_taches(taches)
    
    print "\nCommandes: ajouter, terminer, quitter"
    choix = raw_input("> ").lower()

    if choix == "quitter":
        break
    elif choix == "ajouter":
        nouvelle_tache = raw_input("Nouvelle tache : ")
        if nouvelle_tache:
            taches.append(nouvelle_tache)
            sauvegarder_taches(taches)
    elif choix == "terminer":
        try:
            num_str = raw_input("Numero de la tache a terminer : ")
            num = int(num_str)
            if 1 <= num <= len(taches):
                tache_terminee = taches.pop(num - 1)
                print "Tache '{0}' terminee !".format(tache_terminee)
                sauvegarder_taches(taches)
            else:
                print "Numero de tache invalide."
        except ValueError:
            print "Veuillez entrer un numero valide."
    else:
        print "Commande non reconnue."

print "Sauvegarde des taches et fermeture."
